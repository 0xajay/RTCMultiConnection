    • sessionid | RTCMultiConnection

    It is same as "roomid". It is a readonly property.

    Usage

    var roomid = 'room-id';
connection.open(roomid, function() {
    alert(connection.sesssionid === roomid); // true
});

    Description

    parameterdescription
    sessionid it is a readonly string

    Check if room creator left

    connection.onleave = function(event) {
    if (event.userid === connection.sessionid) {
        alert('Room creator left');
    }
};

connection.onstreamended = function(event) {
    if (event.userid === connection.sessionid) {
        alert('A stream between you and room creator is closed');
    }
};
