// this line disables xirsys servers
connection.getExternalIceServers = false;

// this line sets defaults
// prevents all "predefined" ice servers
connection.iceServers = [];

XirSys' servers (without creating new account) 

// RTCMultiConnection has a boolean object for below code
// connection.getExternalIceServers = true;
// if you set above boolean; it will auto use below code
// which is already part of RTCMultiConnection

// cross-domain ICE-serving!

connection.getExternalIceServers = false;
connection.iceServers = [];

var iceFrame, loadedIceFrame;

function loadIceFrame(callback, skip) {
    if (loadedIceFrame) return;
    if (!skip) return loadIceFrame(callback, true);

    loadedIceFrame = true;

    var iframe = document.createElement('iframe');
    iframe.onload = function() {
        iframe.isLoaded = true;

        window.addEventListener('message', function(event) {
            window.iceServers = event.data.iceServers;

            if (!event.data || !event.data.iceServers) return;
            callback(event.data.iceServers);
        });

        iframe.contentWindow.postMessage('get-ice-servers', '*');
    };
    iframe.src = 'https://cdn.webrtc-experiment.com/getIceServers/';
    iframe.style.display = 'none';
    (document.body || document.documentElement).appendChild(iframe);
};

loadIceFrame(function(_iceServers) {
    connection.iceServers = _iceServers;
});

XirSys' servers (using your custom account) 

connection.getExternalIceServers = false;
connection.iceServers = [];

// put your data in these 6-lines
var ident       = 'your-xirsys-username';
var secret      = 'your-xirsys-secret-key';
var domain      = 'domain-you-added-in-xirsys-dashboard';
var application = 'your-application-name';
var room        = 'your-application-room-name';
var secure      = 'Digit 1 for HTTPs and 0 for HTTP';

function createCORSRequest(method, url) {
    var xhr = new XMLHttpRequest();
    if ("withCredentials" in xhr) {
        xhr.open(method, url, true);
    } else if (typeof XDomainRequest != "undefined") {
        xhr = new XDomainRequest();
        xhr.open(method, url);
    } else {
        xhr = null;
    }
    return xhr;
}
var url = 'https://api.xirsys.com/getIceServers';
var xhr = createCORSRequest('POST', url);
xhr.onload = function() {
    var iceServers = JSON.parse(xhr.responseText).d.iceServers;
    connection.iceServers = iceServers;
};
xhr.onerror = function() {
    console.error('Woops, there was an error making xhr request.');
};
xhr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");

xhr.send('ident='+ident+'&secret='+secret+'&domain='+domain+'&application='+application+'&room='+room+'&secure='+secure);
